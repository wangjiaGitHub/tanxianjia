    function refresh() {
        var sceneBody = window.document.getElementById("sceneBody");
        $.ajax({
            url:'/crazy/finder/game/statistics',
            type:'GET',
            success: function (data) {
                var gameStatistics = JSON.parse(data);
                $("#totalPlayTimes").text(gameStatistics.totalPlayTimes);
                $("#totalDuration").text(gameStatistics.totalDuration);
                $("#totalPlayers").text(gameStatistics.totalPlayers);
                $("#aveDuration").text(gameStatistics.aveDuration);
                $("#gameShareRate").text(gameStatistics.gameShareRate);
                $("#gameShareAcceptRate").text(gameStatistics.gameShareAcceptRate);
                $("#userShareRate").text(gameStatistics.userShareRate);
                $("#userShareAcceptRate").text(gameStatistics.userShareAcceptRate);
                $("#aveUserShareTimes").text(gameStatistics.aveUserShareTimes);

                var sceneStatistics = gameStatistics.sceneStatisticsDTOList
                if (sceneStatistics !== null) {
                    var str = "";
                    for (i in sceneStatistics) {
                        str += "<tr class=\"text-center\">" +
                                "<th class='text-center' scope=\"row\">" + sceneStatistics[i].sceneID + "</th>" +
                                "<td scope=\"row\">" + sceneStatistics[i].sceneName + "</td>" +
                                "<td scope=\"row\">" + sceneStatistics[i].totalPlayTimes + "</td>" +
                                "<td scope=\"row\">" + sceneStatistics[i].totalDuration + "</td>" +
                                "<td scope=\"row\">" + sceneStatistics[i].aveDuration + "</td>" +
                                "</tr>"
                    }
                    sceneBody.innerHTML = str;
                }
            },
            error: function (data) {
                alert("error: " + data);
            }
        })
    }
